        == vacui/result.h header ==

The 'vacui/result.h' header contains the symbols/macro-generators
to manage result types universally.

-------------------------------------------------
        == VACUI_RESULT_DEF_C ==

|
| #define VACUI_RESULT_DEF_C(name, errorType, valueType)
|

The macro generates a new result struct in the following form,
which is C-compatible (VACUI_ABI_EXTERN_C):
        
        struct <name> {
                // Error
                <errorType> error;

                // Value
                <valueType> value;
        
                // Flag, which tell, whether it's an error or not
                bool ok;
        };

The macro also generates these static functions, that make
returning value/error of your result shorter:
        
        // Returns a struct with given value.
        // Sets 'ok' flag to true.
        static struct name <name>Ok(<valueType> value)                                                                      
        {                                                                                                                  
                return (struct <name>){.value = value, .ok = true};                                                                         
        }                                                                                                                 
                
        // Returns a struct instance with given error
        // value.
        //
        // Sets 'ok' flag to false.
        static struct name <name>Error(<errorType> error)                                                                   
        {                                                                                                                 
                return (struct <name>){.error = error, .ok = false};
        }
